Variables & Scope

Variables are containers for storing data values in JavaScript.

Variable Types:
  • var - Function-scoped, can be redeclared
  • let - Block-scoped, can be updated
  • const - Block-scoped, cannot be reassigned
// var - function-scoped
var userName = "Alice";
console.log(userName);

// let - block-scoped
let userAge = 25;
userAge = 26; // Valid

// const - block-scoped, constant
const PI = 3.14159;
// PI = 3.14; // Error - can't reassign

Scope & Hoisting

JavaScript has function scope (var) and block scope (let/const). Variables declared with var are hoisted to the top of their scope.

// Hoisting example
console.log(hoistedVar); // undefined (var is hoisted)
var hoistedVar = "I was hoisted";

// console.log(hoistedLet); // Error (let not hoisted)
let hoistedLet = "I'm not hoisted";

// Scope example
function demoScope() {
  if (true) {
    var functionScoped = "I'm a var";
    let blockScoped = "I'm a let";
  }
  console.log(functionScoped); // Works
  // console.log(blockScoped); // Error
}

Data Types

JavaScript has dynamic types - the same variable can hold different data types.

Primitive Data Types:
  • String - Text values
  • Number - Numeric values
  • Boolean - true/false
  • Null - Intentional empty value
  • Undefined - Variable not initialized
  • Symbol - Unique identifier
  • BigInt - Large integers
// String
let greeting = "Hello World";
console.log(typeof greeting); // "string"

// Number
let integer = 42;
let decimal = 3.14;
console.log(typeof integer); // "number"

// Boolean
let isActive = true;
console.log(typeof isActive); // "boolean"

// Null (intentional empty value)
let emptyValue = null;
console.log(typeof emptyValue); // "object" (historical bug)

// Undefined
let notDefined;
console.log(typeof notDefined); // "undefined"

Reference Types

Objects, arrays and functions are reference types in JavaScript.

// Object
let person = {
  name: "John",
  age: 30
};
console.log(typeof person); // "object"

// Array
let colors = ["red", "green", "blue"];
console.log(typeof colors); // "object"

// Function
function greet() { return "Hello"; }
console.log(typeof greet); // "function"

Operators

Operators perform operations on variables and values.

Operator Types:
  • Arithmetic - +, -, *, /, %, **
  • Assignment - =, +=, -=, *=, /=
  • Comparison - ==, ===, !=, !==, >, <, >=, <=
  • Logical - &&, ||, !
// Arithmetic
let x = 10, y = 3;
console.log("10 + 3 =", x + y); // 13
console.log("10 % 3 =", x % y); // 1 (remainder)
console.log("10 ** 3 =", x ** y); // 1000 (exponent)

// Assignment
let counter = 5;
counter += 2; // counter = counter + 2
console.log(counter); // 7

// Comparison
console.log("10 == '10':", x == '10'); // true (loose equality)
console.log("10 === '10':", x === '10'); // false (strict equality)

// Logical
console.log("true && false:", true && false); // false
console.log("true || false:", true || false); // true
console.log("!true:", !true); // false

Functions

Functions are reusable blocks of code that perform specific tasks.

Function Types:
  • Function Declarations - Hoisted, can be called before declaration
  • Function Expressions - Stored in variables, not hoisted
  • Arrow Functions - Shorter syntax, no 'this' binding
// Function Declaration
function greet(name) {
  return `Hello, ${name}!`;
}
console.log(greet("Alice"));

// Function Expression
const multiply = function(a, b) {
  return a * b;
};
console.log(multiply(4, 6));

// Arrow Function
const square = num => num * num;
console.log(square(5));

// Default Parameters
function createUser(name, role = "user") {
  return `${name} is a ${role}`;
}
console.log(createUser("Bob")); // "Bob is a user"

Control Flow

Control flow statements determine the order in which code executes.

Conditional Statements

// if/else
function checkAge(age) {
  if (age < 13) {
    return "Child";
  } else if (age < 20) {
    return "Teenager";
  } else {
    return "Adult";
  }
}

// Ternary Operator (shorthand if/else)
const isAdult = (age) => age >= 18 ? "Adult" : "Minor";

// Switch Statement
function getDayName(dayNum) {
  switch(dayNum) {
    case 1: return "Monday";
    case 2: return "Tuesday";
    // ... other cases
    default: return "Invalid day";
  }
}

Loops

// for loop
for (let i = 0; i < 5; i++) {
  console.log(i);
}

// while loop
let count = 0;
while (count < 5) {
  console.log(count);
  count++;
}

// for...of loop (arrays)
const colors = ["red", "green", "blue"];
for (const color of colors) {
  console.log(color);
}

// for...in loop (objects)
const person = {name: "John", age: 30};
for (const key in person) {
  console.log(`${key}: ${person[key]}`);
}

Arrays

Arrays are ordered collections of values that can hold multiple data types.

Common Array Methods:
  • push/pop - Add/remove from end
  • shift/unshift - Add/remove from start
  • slice/splice - Extract/insert elements
  • map/filter/reduce - Transform arrays
// Creating arrays
const fruits = ["apple", "banana", "orange"];
const numbers = [1, 2, 3, 4, 5];

// Adding/removing elements
fruits.push("mango"); // Add to end
fruits.pop(); // Remove from end
fruits.unshift("strawberry"); // Add to start
fruits.shift(); // Remove from start

// Iteration methods
const doubled = numbers.map(num => num * 2);
const evens = numbers.filter(num => num % 2 === 0);
const sum = numbers.reduce((total, num) => total + num, 0);

// Finding elements
const hasApple = fruits.includes("apple");
const bananaIndex = fruits.indexOf("banana");

Objects

Objects store collections of key-value pairs and more complex entities.

Object Features:
  • Properties - Key-value pairs
  • Methods - Functions as properties
  • this - Refers to the object
  • Destructuring - Extract properties
// Object creation
const person = {
  firstName: "John",
  lastName: "Doe",
  age: 30,
  hobbies: ["reading", "coding"],
  fullName() {
    return `${this.firstName} ${this.lastName}`;
  }
};

// Accessing properties
console.log(person.firstName); // Dot notation
console.log(person["lastName"]); // Bracket notation

// Adding/modifying
person.email = "john@example.com";
person.age = 31;

// Object methods
console.log(person.fullName());

// Object iteration
for (const key in person) {
  console.log(`${key}: ${person[key]}`);
}

// Destructuring
const { firstName, age } = person;
console.log(firstName, age);

DOM Manipulation

The Document Object Model (DOM) represents the page as objects that JavaScript can manipulate.

Key DOM Concepts:
  • Selecting Elements - getElementById, querySelector, etc.
  • Modifying Content - textContent, innerHTML
  • Changing Styles - style property, classList
  • Event Handling - addEventListener
// Selecting elements
const heading = document.getElementById('main-heading');
const boxes = document.getElementsByClassName('box');
const firstDiv = document.querySelector('div');

// Modifying content
heading.textContent = "New Heading"; // Text only
firstDiv.innerHTML = "<strong>Bold text</strong>"; // Parses HTML

// Changing styles
heading.style.color = "blue";
heading.classList.add("highlight");

// Event handling
const button = document.querySelector('button');
button.addEventListener('click', function() {
  console.log("Button clicked!");
});

Events

Events are actions that happen in the browser that JavaScript can respond to.

Common Event Types:
  • Mouse Events - click, mouseover, mouseout
  • Keyboard Events - keydown, keyup
  • Form Events - submit, change, focus
  • Window Events - load, resize, scroll
// Click event
button.addEventListener('click', function(e) {
  console.log("Clicked!", e.target);
});

// Form submission
form.addEventListener('submit', function(e) {
  e.preventDefault(); // Prevent page reload
  console.log("Form submitted");
});

// Keyboard input
input.addEventListener('keydown', function(e) {
  console.log("Key pressed:", e.key);
});

// Mouse events
box.addEventListener('mouseover', function() {
  this.style.backgroundColor = "yellow";
});

box.addEventListener('mouseout', function() {
  this.style.backgroundColor = "";
});

// Event delegation
document.getElementById('list').addEventListener('click', function(e) {
  if (e.target.tagName === 'LI') {
    console.log("List item clicked:", e.target.textContent);
  }
});

Summary

1 JavaScript has variables (var, let, const), data types, and operators

2 Functions are reusable code blocks (declarations, expressions, arrow)

3 Control flow includes conditionals (if/else, switch) and loops (for, while)

4 Arrays store ordered collections, objects store key-value pairs

5 DOM manipulation lets you interact with and modify web pages
